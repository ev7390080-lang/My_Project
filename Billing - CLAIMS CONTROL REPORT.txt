SELECT TD.TXN_DETAIL_ID,
(SELECT MAX(PER_ID_NBR) FROM CI_PER_ID PERID, CI_ACCT_PER AP,CI_PER_PER PERPER
WHERE PRITM.ACCT_ID =AP.ACCT_ID
AND PERPER.PER_ID2 =AP.PER_ID
AND PERPER.PER_ID1 = PERID.PER_ID
AND PERID.ID_TYPE_CD = 'CLIENTID') AS CLIENT_ID,
TD.TXN_DTTM CLAIM_RECD_DT,
TD.TXN_VOL AS PAID_COUNT,
0 AS REVERSED_COUNT,
0 AS DENIED_COUNT,
TD.UDF_CHAR_18 AS CLAIM_STATUS,
TD.UDF_CHAR_31 AS Claim_Type,
TD.TXN_AMT AS PAID_TOTAL_AMT,
0 AS REVERSED_TOTAL_AMT,
0 AS DENIED_TOTAL_AMT,
(SELECT ALT_BILL_ID FROM CI_BILL BILL, CI_BSEG BSEG, CI_BSEG_CALC CALC
WHERE BILL.BILL_ID = BSEG.BILL_ID
AND BSEG.BSEG_STAT_FLG ='50'
AND BSEG.BSEG_ID =CALC.BSEG_ID
AND CALC.BILLABLE_CHG_ID =CHG.BILLABLE_CHG_ID)AS INVOICE_NUM,
(SELECT BILL_DT FROM CI_BILL BILL, CI_BSEG BSEG, CI_BSEG_CALC CALC
WHERE BILL.BILL_ID = BSEG.BILL_ID
AND BSEG.BSEG_STAT_FLG ='50'
AND BSEG.BSEG_ID =CALC.BSEG_ID
AND CALC.BILLABLE_CHG_ID =CHG.BILLABLE_CHG_ID)AS BILL_DT
FROM CI_TXN_DETAIL TD,CI_TXN_DTL_PRITM PRITM,CI_BILL_CHG CHG
WHERE 1=1
AND TD.BO_STATUS_CD = 'COMP'
AND TD.TXN_REC_TYPE_CD= 'CLIENTBILLING'
AND TD.TXN_DETAIL_ID = PRITM.TXN_DETAIL_ID                             
AND PRITM.BILLABLE_CHG_ID =CHG.BILLABLE_CHG_ID
AND CHG.PRICEITEM_CD =PRITM.INITIAL_PRICE_ITEM_CD
AND UDF_CHAR_18 ='P'
UNION
SELECT TD.TXN_DETAIL_ID,
(SELECT MAX(PER_ID_NBR) FROM CI_PER_ID PERID, CI_ACCT_PER AP,CI_PER_PER PERPER
WHERE PRITM.ACCT_ID =AP.ACCT_ID
AND PERPER.PER_ID2 =AP.PER_ID
AND PERPER.PER_ID1 = PERID.PER_ID
AND PERID.ID_TYPE_CD = 'CLIENTID') AS CLIENT_ID,
TD.TXN_DTTM CLAIM_RECD_DT,
0 AS PAID_COUNT,
TD.TXN_VOL AS REVERSED_COUNT,
0 AS DENIED_COUNT,
TD.UDF_CHAR_18 AS CLAIM_STATUS,
TD.UDF_CHAR_31 AS Claim_Type,
0 AS PAID_TOTAL_AMT,
TD.TXN_AMT*-1 AS REVERSED_TOTAL_AMT,
0 AS DENIED_TOTAL_AMT,
(SELECT ALT_BILL_ID FROM CI_BILL BILL, CI_BSEG BSEG, CI_BSEG_CALC CALC
WHERE BILL.BILL_ID = BSEG.BILL_ID
AND BSEG.BSEG_STAT_FLG ='50'
AND BSEG.BSEG_ID =CALC.BSEG_ID
AND CALC.BILLABLE_CHG_ID =CHG.BILLABLE_CHG_ID)AS INVOICE_NUM,
(SELECT BILL_DT FROM CI_BILL BILL, CI_BSEG BSEG, CI_BSEG_CALC CALC
WHERE BILL.BILL_ID = BSEG.BILL_ID
AND BSEG.BSEG_STAT_FLG ='50'
AND BSEG.BSEG_ID =CALC.BSEG_ID
AND CALC.BILLABLE_CHG_ID =CHG.BILLABLE_CHG_ID)AS BILL_DT
FROM CI_TXN_DETAIL TD,CI_TXN_DTL_PRITM PRITM,CI_BILL_CHG CHG
WHERE 1=1
AND TD.BO_STATUS_CD = 'COMP'
AND TD.TXN_REC_TYPE_CD= 'CLIENTBILLING'
AND TD.TXN_DETAIL_ID = PRITM.TXN_DETAIL_ID                             
AND PRITM.BILLABLE_CHG_ID =CHG.BILLABLE_CHG_ID
AND CHG.PRICEITEM_CD =PRITM.INITIAL_PRICE_ITEM_CD
AND UDF_CHAR_18 ='X'
UNION
SELECT TD.TXN_DETAIL_ID,
(SELECT MAX(PER_ID_NBR) FROM CI_PER_ID PERID, CI_ACCT_PER AP,CI_PER_PER PERPER
WHERE PRITM.ACCT_ID =AP.ACCT_ID
AND PERPER.PER_ID2 =AP.PER_ID
AND PERPER.PER_ID1 = PERID.PER_ID
AND PERID.ID_TYPE_CD = 'CLIENTID') AS CLIENT_ID,
TD.TXN_DTTM CLAIM_RECD_DT,
0 AS PAID_COUNT,
0 AS REVERSED_COUNT,
TD.TXN_VOL AS DENIED_COUNT,
TD.UDF_CHAR_18 AS CLAIM_STATUS,
TD.UDF_CHAR_31 AS Claim_Type,
0 AS PAID_TOTAL_AMT,
0 AS REVERSED_TOTAL_AMT,
TD.TXN_AMT AS DENIED_TOTAL_AMT,
(SELECT ALT_BILL_ID FROM CI_BILL BILL, CI_BSEG BSEG, CI_BSEG_CALC CALC
WHERE BILL.BILL_ID = BSEG.BILL_ID
AND BSEG.BSEG_STAT_FLG ='50'
AND BSEG.BSEG_ID =CALC.BSEG_ID
AND CALC.BILLABLE_CHG_ID =CHG.BILLABLE_CHG_ID)AS INVOICE_NUM,
(SELECT BILL_DT FROM CI_BILL BILL, CI_BSEG BSEG, CI_BSEG_CALC CALC
WHERE BILL.BILL_ID = BSEG.BILL_ID
AND BSEG.BSEG_STAT_FLG ='50'
AND BSEG.BSEG_ID =CALC.BSEG_ID
AND CALC.BILLABLE_CHG_ID =CHG.BILLABLE_CHG_ID)AS BILL_DT
FROM CI_TXN_DETAIL TD,CI_TXN_DTL_PRITM PRITM,CI_BILL_CHG CHG
WHERE 1=1
AND TD.BO_STATUS_CD = 'COMP'
AND TD.TXN_REC_TYPE_CD= 'CLIENTBILLING'
AND TD.TXN_DETAIL_ID = PRITM.TXN_DETAIL_ID                             
AND PRITM.BILLABLE_CHG_ID =CHG.BILLABLE_CHG_ID
AND CHG.PRICEITEM_CD =PRITM.INITIAL_PRICE_ITEM_CD
AND UDF_CHAR_18 ='R'
